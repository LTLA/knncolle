144  template<class PAIRED>
145  static std::vector<std::pair<INDEX_t, DISTANCE_t> > reformat(const PAIRED& results, bool check_self = false, INDEX_t self_index = 0) {
146  // If check_self=false, then we pretend that the self is already
147  // found, in which case the two clauses below are just skipped.
148  bool self_found = !check_self;
149 
150  std::vector<std::pair<INDEX_t, DISTANCE_t> > output;
151  for (const auto& current : results) {
152  if (!self_found && current.second == self_index) {
153  self_found=true;
154  continue;
155  } else {
156  output.push_back(std::pair<INDEX_t, DISTANCE_t>(current.second, current.first));
157  }
158  }
159 
160  // Just in case we're full of ties at duplicate points, such that 'c'
161  // is not in the set. Note that, if self_found=false, we must have at
162  // least 'K+2' points for 'c' to not be detected as its own neighbor.
163  // Thus there is no need to worry whether we are popping off a non-'c'
164  // element at the end of the vector.
165  if (!self_found) {
166  output.pop_back();
167  }
168  return output;
169  }
