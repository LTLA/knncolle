111 std::vector<std::pair<INDEX_t, DISTANCE_t> > find_nearest_neighbors(INDEX_t index, int k) const {
112 std::vector<INTERNAL_INDEX_t> indices;
113 std::vector<INTERNAL_DATA_t> distances;
114 annoy_index.get_nns_by_item(index, k + 1, get_search_k(k + 1), &indices, &distances); // +1, as it forgets to discard 'self'.
115
116 bool self_found = false;
117 const INTERNAL_INDEX_t self = index;
118 std::vector<std::pair<INDEX_t, DISTANCE_t> > output;
119 output.reserve(k);
120 for (size_t i = 0; i < indices.size(); ++i) {
121 if (!self_found && indices[i] == self) {
122 self_found=true;
123 } else {
124 output.emplace_back(indices[i], distances[i]);
125 }
126 }
127
128 // Just in case we're full of ties at duplicate points, such that 'c'
129 // is not in the set. Note that, if self_found=false, we must have at
130 // least 'K+2' points for 'c' to not be detected as its own neighbor.
131 // Thus there is no need to worry whether we are popping off a non-'c'
132 // element at the end of the vector.
133 if (!self_found) {
134 output.pop_back();
135 }
136
137 return output;
138 }

140 std::vector<std::pair<INDEX_t, DISTANCE_t> > find_nearest_neighbors(const QUERY_t* query, int k) const {
141 std::vector<INTERNAL_INDEX_t> indices;
142 indices.reserve(k);
143 std::vector<INTERNAL_DATA_t> distances;
144 distances.reserve(k);
145
146 if constexpr(std::is_same<INTERNAL_DATA_t, QUERY_t>::value) {
147 annoy_index.get_nns_by_vector(query, k, get_search_k(k), &indices, &distances);
148 } else {
149 std::vector<INTERNAL_DATA_t> tmp(query, query + num_dim);
150 annoy_index.get_nns_by_vector(tmp.data(), k, get_search_k(k), &indices, &distances);
151 }
152
153 std::vector<std::pair<INDEX_t, DISTANCE_t> > output;
154 output.reserve(k);
155 for (size_t i = 0; i < indices.size(); ++i) {
156 output.emplace_back(indices[i], distances[i]);
157 }
158 return output;
159 }
