80 template<typename INPUT_t>
81 Kmknn(INDEX_t ndim, INDEX_t nobs, const INPUT_t* vals, double power = 0.5) :
82 num_dim(ndim),
83 num_obs(nobs),
84 data(ndim * nobs),
85 sizes(std::ceil(std::pow(num_obs, power))),
86 offsets(sizes.size()),
87 centers(sizes.size() * ndim),
88 observation_id(nobs),
89 new_location(nobs),
90 dist_to_centroid(nobs)
91 {
92 std::vector<int> clusters(num_obs);
93 auto ncenters = sizes.size();
94 auto output = kmeans::Kmeans<INTERNAL_t, int>().run(ndim, nobs, vals, ncenters, centers.data(), clusters.data());
95 std::swap(sizes, output.sizes);
96
97 // In case there were some duplicate points, we just resize this a bit.
98 if (ncenters != sizes.size()) {
99 ncenters = sizes.size();
100 offsets.resize(ncenters);
101 centers.resize(ncenters * ndim);
102 }
103
104 for (INDEX_t i = 1; i < ncenters; ++i) {
105 offsets[i] = offsets[i - 1] + sizes[i - 1];
106 }
107
108 // Organize points correctly; firstly, sorting by distance from the assigned center.
109 std::vector<std::pair<INTERNAL_t, INDEX_t> > by_distance(nobs);
110 {
111 auto sofar = offsets;
112 for (INDEX_t o = 0; o < nobs; ++o) {
113 const auto& clustid = clusters[o];
114 auto& counter = sofar[clustid];
115 auto& current = by_distance[counter];
116 current.first = DISTANCE::normalize(DISTANCE::template raw_distance<INTERNAL_t>(vals + o * num_dim, centers.data() + clustid * num_dim, num_dim));
117 current.second = o;
118 ++counter;
119 }
120
121 for (INDEX_t c = 0; c < ncenters; ++c) {
122 auto begin = by_distance.begin() + offsets[c];
123 std::sort(begin, begin + sizes[c]);
124 }
125 }
126
127 // Now, copying this over.
128 {
129 auto store = data.data();
130 for (INDEX_t o = 0; o < nobs; ++o, store += num_dim) {
131 const auto& current = by_distance[o];
132 auto source = vals + ndim * current.second;
133 std::copy(source, source + ndim, store);
134 observation_id[o] = current.second;
135 new_location[current.second] = o;
136 dist_to_centroid[o] = current.first;
137 }
138 }
139
140 return;
141 }

169 template<typename INPUT_t>
170 void search_nn(INPUT_t* target, NeighborQueue<INDEX_t, INTERNAL_t>& nearest) const {
171 /* Computing distances to all centers and sorting them. The aim is to
172 * go through the nearest centers first, to get the shortest
173 * 'threshold' possible through the rest of the search.
174 */
175 std::vector<std::pair<INTERNAL_t, INDEX_t> > center_order(sizes.size());
176 auto clust_ptr = centers.data();
177 for (size_t c = 0; c < sizes.size(); ++c, clust_ptr += num_dim) {
178 center_order[c].first = DISTANCE::template raw_distance<INTERNAL_t>(target, clust_ptr, num_dim);
179 center_order[c].second = c;
180 }
181 std::sort(center_order.begin(), center_order.end());
182 INTERNAL_t threshold_raw = -1;
183
184 // Computing the distance to each center, and deciding whether to proceed for each cluster.
185 for (const auto& curcent : center_order) {
186 const INDEX_t center = curcent.second;
187 const INTERNAL_t dist2center = DISTANCE::normalize(curcent.first);
188
189 const auto cur_nobs = sizes[center];
190 const double* dIt = dist_to_centroid.data() + offsets[center];
191 const double maxdist = *(dIt + cur_nobs - 1);
192
193 INDEX_t firstcell=0;
194#if USE_UPPER
195 double upper_bd = std::numeric_limits<INTERNAL_t>::max();
196#endif
197
198 if (threshold_raw >= 0) {
199 const INTERNAL_t threshold = DISTANCE::normalize(threshold_raw);
200
201 /* The conditional expression below exploits the triangle inequality; it is equivalent to asking whether:
202 * threshold + maxdist < dist2center
203 * All points (if any) within this cluster with distances above 'lower_bd' are potentially countable.
204 */
205 const double lower_bd = dist2center - threshold;
206 if (maxdist < lower_bd) {
207 continue;
208 }
209 firstcell=std::lower_bound(dIt, dIt + cur_nobs, lower_bd) - dIt;
210#if USE_UPPER
211 /* This exploits the reverse triangle inequality, to ignore points where:
212 * threshold + dist2center < point-to-center distance
213 */
214 upper_bd = threshold + dist2center;
215#endif
216 }
217
218 const auto cur_start = offsets[center];
219 const double* other_cell = data.data() + num_dim * (cur_start + firstcell);
220 for (auto celldex = firstcell; celldex < cur_nobs; ++celldex, other_cell += num_dim) {
221#if USE_UPPER
222 if (*(dIt + celldex) > upper_bd) {
223 break;
224 }
225#endif
226 const auto dist2cell_raw = DISTANCE::template raw_distance<INTERNAL_t>(target, other_cell, num_dim);
227 nearest.add(cur_start + celldex, dist2cell_raw);
228 if (nearest.is_full()) {
229 threshold_raw = nearest.limit(); // Shrinking the threshold, if an earlier NN has been found.
230#if USE_UPPER
231 upper_bd = DISTANCE::normalize(threshold_raw) + dist2center;
232#endif
233 }
234 }
235 }
236 }
