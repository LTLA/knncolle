template<class ANNOY_DISTANCE, typename ITYPE = int, typename DTYPE = double>
class knncolle::AnnoyBase< ANNOY_DISTANCE, ITYPE, DTYPE >

Perform an approximate nearest neighbor search with Annoy.

Template Parameters
