93  static void purge_self(ITYPE self_index, std::vector<ITYPE>* indices, std::vector<DTYPE>* distances) {
94  bool self_found=false;
95  auto& current = *indices;
96  for (size_t idx=0; idx < current.size(); ++idx) {
97  if (current[idx] == self_index) {
98  current.erase(current.begin() + idx);
99  if (distances) {
100  distances->erase(distances->begin() + idx);
101  }
102  self_found=true;
103  break;
104  }
105  }
106 
107  // Just in case we're full of ties at duplicate points, such that 'c' is not in the set.
108  // Note that, if self_found=false, we must have at least 'K+2' points for 'c' to not
109  // be detected as its own neighbor. Thus there is no need to worry whether we are popping
110  // off a non-'c' element at the end of the vector.
111  if (!self_found) {
112  current.pop_back();
