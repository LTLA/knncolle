67 // Single node of a VP tree (has a point and radius; left children are closer to point than the radius)
68 struct Node {
69 INTERNAL_t threshold; // radius
70 INDEX_t index; // original index of current vantage point
71 NodeIndex_t left; // node index of the next vantage point for all children closer than 'threshold' from the current vantage point
72 NodeIndex_t right; // node index of the next vantage point for all children further than 'threshold' from the current vantage point
73 Node(NodeIndex_t i=0) : threshold(0), index(i), left(LEAF_MARKER), right(LEAF_MARKER) {}
74 };

79 template<class SAMPLER>
80 NodeIndex_t buildFromPoints(NodeIndex_t lower, NodeIndex_t upper, std::vector<DataPoint>& items, SAMPLER& rng) {
81 if (upper == lower) { // indicates that we're done here!
82 return LEAF_MARKER;
83 }
84
85 NodeIndex_t pos = nodes.size();
86 nodes.resize(pos + 1);
87 Node& node=nodes.back();
88
89 int gap = upper - lower;
90 if (gap > 1) { // if we did not arrive at leaf yet
91
92 /* Choose an arbitrary point and move it to the start of the [lower, upper)
93 * interval in 'items'; this is our new vantage point.
94 *
95 * Yes, I know that the modulo method does not provide strictly
96 * uniform values but statistical correctness doesn't really matter
97 * here... but reproducibility across platforms does matter, and
98 * std::uniform_int_distribution is implementation-dependent!
99 */
100 NodeIndex_t i = static_cast<NodeIndex_t>(rng() % gap + lower);
101 std::swap(items[lower], items[i]);
102 const auto& vantage = items[lower];
103
104 // Compute distances to the new vantage point.
105 const INTERNAL_t* ref = std::get<1>(vantage);
106 for (size_t i = lower + 1; i < upper; ++i) {
107 const INTERNAL_t* loc = std::get<1>(items[i]);
108 std::get<2>(items[i]) = DISTANCE::template raw_distance<INTERNAL_t>(ref, loc, num_dim);
109 }
110
111 // Partition around the median distance from the vantage point.
112 NodeIndex_t median = lower + gap/2;
113 std::nth_element(items.begin() + lower + 1, items.begin() + median, items.begin() + upper,
114 [&](const DataPoint& left, const DataPoint& right) -> bool {
115 return std::get<2>(left) < std::get<2>(right);
116 }
117 );
118
119 // Threshold of the new node will be the distance to the median
120 node.threshold = DISTANCE::normalize(std::get<2>(items[median]));
121
122 // Recursively build tree
123 node.index = std::get<0>(vantage);
124 node.left = buildFromPoints(lower + 1, median, items, rng);
125 node.right = buildFromPoints(median, upper, items, rng);
126 } else {
127 node.index = std::get<0>(items[lower]);
128 }
129
130 return pos;
131 }

147 template<typename INPUT_t>
148 VpTree(INDEX_t ndim, INDEX_t nobs, const INPUT_t* vals, bool copy = false) : num_dim(ndim), num_obs(nobs), store(vals, ndim * nobs, copy) {
149 std::vector<DataPoint> items;
150 items.reserve(num_obs);
151 auto ptr = store.reference;
152 for (INDEX_t i = 0; i < num_obs; ++i, ptr += num_dim) {
153 items.push_back(DataPoint(i, ptr, 0));
154 }
155
156 nodes.reserve(num_obs);
157 std::mt19937_64 rand(1234567890); // seed doesn't really matter, we don't need statistical correctness here.
158 buildFromPoints(0, num_obs, items, rand);
159 return;
160 }

189 template<typename INPUT_t>
190 void search_nn(NodeIndex_t curnode_index, const INPUT_t* target, INTERNAL_t& tau, NeighborQueue<INDEX_t, INTERNAL_t>& nearest) const {
191 if (curnode_index == LEAF_MARKER) { // indicates that we're done here
192 return;
193 }
194
195 // Compute distance between target and current node
196 const auto& curnode=nodes[curnode_index];
197 INTERNAL_t dist = DISTANCE::normalize(DISTANCE::template raw_distance<INTERNAL_t>(store.reference + curnode.index * num_dim, target, num_dim));
198
199 // If current node within radius tau
200 if (dist < tau) {
201 nearest.add(curnode.index, dist);
202 if (nearest.is_full()) {
203 tau = nearest.limit(); // update value of tau (farthest point in result list)
204 }
205 }
206
207 // Return if we arrived at a leaf
208 if (curnode.left == LEAF_MARKER && curnode.right == LEAF_MARKER) {
209 return;
210 }
211
212 // If the target lies within the radius of ball
213 if (dist < curnode.threshold) {
214 if (dist - tau <= curnode.threshold) { // if there can still be neighbors inside the ball, recursively search left child first
215 search_nn(curnode.left, target, tau, nearest);
216 }
217
218 if (dist + tau >= curnode.threshold) { // if there can still be neighbors outside the ball, recursively search right child
219 search_nn(curnode.right, target, tau, nearest);
220 }
221
222 // If the target lies outsize the radius of the ball
223 } else {
224 if (dist + tau >= curnode.threshold) { // if there can still be neighbors outside the ball, recursively search right child first
225 search_nn(curnode.right, target, tau, nearest);
226 }
227
228 if (dist - tau <= curnode.threshold) { // if there can still be neighbors inside the ball, recursively search left child
229 search_nn(curnode.left, target, tau, nearest);
230 }
231 }
232 }
